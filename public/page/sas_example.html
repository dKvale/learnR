The R script

R Setup

# Load helper packages to connect to data lake
library(mdhdb)
library(tidyverse)
library(DBI)
library(sqldf)


# Connect to data lake
miic <- open_mncloud("miic", schema = "")

Get C19 immunization data

R code

# After Jan 1st, 2022
all_vaccinations <- miic %>% 
                    tbl("immunization") %>% 
                    filter(date_entered > 1641000022400)

vaccine_groups   <- miic %>% 
                    tbl("vaccine_group_children") %>% 
                    select(vaccine_id, vaccine_group_id) 


# Join group_ids and keep only COVID
c19_vax <- all_vaccinations %>% 
           left_join(vaccine_groups) %>% 
           filter(vaccine_group_id == 48)

SQL in R

immunization <- dbGetQuery(miic, 
     "SELECT *
      FROM IMMUNIZATION t1
           INNER JOIN (SELECT VACCINE_ID as vax_id, VACCINE_GROUP_ID FROM VACCINE_GROUP_CHILDREN WHERE VACCINE_GROUP_ID = 48) t2 ON (t1.VACCINE_ID = t2.vax_id)
      WHERE t1.DATE_ENTERED > 1641000022400
      LIMIT 50")


Add CVX (Vaccine ID)

R code

vaccine_ids <- miic %>% tbl("vaccine") 

c19_vax <- c19_vax %>% left_join(vaccine_ids %>% select(vaccine_id, cvx_code))

SQL in R

vaccine_ids <- dbGetQuery(miic, "SELECT vaccine_id as vaccine_id_vax, cvx_code FROM VACCINE")

c19_vax <- sqldf("SELECT * FROM immunization t1
                   LEFT JOIN vaccine_ids t2 ON (t1.vaccine_id = t2.vaccine_id_vax)")

Add MVX (Manufacturer)

R code

manufacturer <- miic %>% tbl("manufacturer") %>% select(manufacturer_id, manufacturer_code)

c19_vax <- c19_vax %>% left_join(manufacturer)

SQL in R

manufacturers <- dbGetQuery(miic,  "SELECT manufacturer_id as manuf_id2, manufacturer_code FROM MANUFACTURER")

c19_vax <- sqldf("SELECT * FROM c19_vax t1
                  LEFT JOIN manufacturers t2 ON (t1.MANUFACTURER_ID = t2.manuf_id2)")


Join more tables

More table joining > 
# Add lot number
## MIICPRD.VACCINE_LOT table join vaccine_lot_number joins on vaccine_lot 
lot <- miic %>% tbl("vaccine_lot") %>% select(vaccine_lot_id, lot_number)

c19_vax <- c19_vax %>% left_join(vaccine_lot)


# Collect the data
c19 <- c19_vax %>% head(n = 20000) %>% select(-contains("hoodie"), -change_seq) %>% collect()


# Dose number
dose <- miic %>% 
        tbl("immun_fact") %>% 
        filter(vaccine_group == 48, entered_system > 1641000022400) %>%
        select(immunization_id, ordinal_position, number_in_series, validity) %>% 
        head(n = 10000) %>% 
        collect()

c19 <- c19 %>% left_join(dose)


# Provider
org <- miic %>% 
       tbl("provider_organization") %>% 
       select(provider_organization_id, address_id, name, mnvfc_pin, organization_type, subtype, fqhc_rhc, tribal_ind) %>% 
       collect()

c19 <- c19 %>% left_join(org)


# Provider address
org_address <- miic %>% tbl("address") %>% select(address_id, street_address_line, other_address_line, city_name, county_code, state_code, zip_code) %>% head(n = 1000) %>% collect()

c19 <- c19 %>% left_join(org_address)


# Provider county
countys <- miic %>% tbl("county") %>% select(county_code, fips) %>% collect()

c19 <- c19 %>% left_join(countys)


# Client details
client <- miic %>% tbl("client") %>% select(client_id, first_name:birth_date, sex_code, race_code) %>% head(n = 20) %>% collect()

c19 <- c19 %>% left_join(client)


# Client address
client_address <- miic %>% 
                  tbl("person_address") %>% 
                  select(client_id:county_code) %>% 
                  rename_at(vars(-client_id), ~paste0(., "_c")) %>% 
                  head(n = 20) %>% 
                  collect()

c19 <- c19 %>% left_join(client_address)


# Client FIPS
client_fips <- countys %>% rename_all(~paste0(., "_c")) %>% collect()

c19 <- c19 %>% left_join(client_fips %>% select(county_code_c, fips_c))


# Collect
c19 <- c19 %>% collect()

An alternative

You can store a pre-generated view in the data lake for complex queries that need to be run frequently.

c19 <- miic %>% tbl("miic_vaccine_export_vw")


Final formatting

Rename columns

c19 <- c19 %>%
       rename(vax_event_id        = immunization_id,
              recip_id            = client_id,
              recip_first_name    = first_name,
              recip_dob           = birth_date,
              admin_address_state = state_code,
              admin_address_zip   = zip_code)

Redact names

## Redact
c19 <- c19 %>%
       mutate(recip_first_name  = "Redacted",
              recip_middle_name = "Redacted",
              recip_last_name   = "Redacted")

Add new columns

c19 <- c19 %>%
       mutate(pprl_id  = "",
              ext_type = "D",
              admin_address_county = paste("27", fips), 
              recip_address_county = paste("27", fips_c), 
              vax_refusal = "No", 
              serology = "UNK",
              cmorbid_status = "UNK",
              vax_series_complete = case_when(
                                       ordinal_position > 0 & ordinal_position == number_in_series ~ "YES",
                                       ordinal_position > 0 & ordinal_position < number_in_series ~ "NO"))

Add CDC codes

## Add CDC admin_type codes
c19 <- c19 %>% left_join(cdc_adminType_codes)

## Add CDC race codes
c19 <- c19 %>% left_join(cdc_race_codes)
      
## Add CDC vax_route codes
c19 <- c19 %>% left_join(cdc_vaxRoute_codes)

Filter to only MN

c19 <- c19 %>%
       filter(!admin_address_state %in% c("ND", "SD", "IA","WI"))

# Filter providers
c19 <- c19 %>%
       filter(!provider_organization_id %in% c(4472, 17190, 41592, 532, 36240))


Organize final table

# Organize table columns and arrange by Vax Event
c19 <- c19 %>%
       select(vax_event_id, pprl_id, ext_type, recip_id, recip_first_name, everything()) %>%
       arrange(vax_event_id)
